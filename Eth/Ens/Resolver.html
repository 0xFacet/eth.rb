def namehash(ens_name)
  node = ("0" * 64)
  name = normalize(ens_name)
  name.split(".").reverse.each do |label|
    hash = Digest::Keccak.new(256).hexdigest(label)
    node = Digest::Keccak.new(256).hexdigest([node + hash].pack("H*"))
  end
  "0x#{node}"
end
